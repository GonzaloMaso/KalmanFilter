12 StaticROUKF::StaticROUKF(int nObservations, int nStates, int nParameters, double* observationsUncertainty,
13  double* parametersUncertainty, SigmaPointsGenerator::SIGMA_DISTRIBUTION sigmaDistribution) {
14 
15  this->nObservations = nObservations;
16  this->nStates = nStates;
17  this->nParameters = nParameters;
18 
19  Theta = zeros(nParameters, 1);
20 
21  LTheta = eye(nParameters, nParameters);
22  U = eye(nParameters, nParameters);
23  Wi = eye(nObservations, nObservations);
24 
25  mat diagParameter(parametersUncertainty, 1, nParameters);
26  U.diag() = 1. / diagParameter;
27  mat diagObservations(observationsUncertainty, 1, nObservations);
28  Wi.diag() = 1. / diagObservations;
29 
30  SigmaPointsGenerator::generateSigmaPoints(nParameters,sigmaDistribution,&sigma);
31  alpha = 1. / sigma.n_cols;
32

39 double StaticROUKF::executeStep(double *zkhatc, forwardOp A, observationOp H) {
40 
41  // Matrixes
42  mat Thetak(nParameters, sigma.n_cols), Xk(nStates, sigma.n_cols), Zk(
43  nObservations, sigma.n_cols);
44  mat HL;
45  mat invU = inv(U);
46  mat C = chol(invU);
47 
48  // Column vectors
49  mat thetak, zk;
50  mat zkhat(zkhatc, nObservations, 1);
51 
52  double *thetakdata, *xkdata, *zkdata;
53  thetakdata = new double[nParameters];
54  xkdata = new double[nStates];
55  zkdata = new double[nObservations];
56 
57  for (unsigned int i = 0; i < sigma.n_cols; i++) {
58  // Sampling
59  mat s = sigma.col(i);
60  thetak = Theta + LTheta * C.t() * s;
61 
62  // Propagate sigma point
63  // Copies xk to a primitive type in order to achieve armadillo independence in extern programs
64  memcpy(thetakdata, thetak.memptr(), nParameters * sizeof(double));
65 
66  (*A)(xkdata, nStates, thetakdata, nParameters);
67  thetak = mat(thetakdata, nParameters, 1);
68 
69  // Perform observation
70  (*H)(xkdata, nStates, zkdata, nObservations);
71  zk = mat(zkdata, nObservations, 1);
72 
73  Thetak.col(i) = thetak;
74  Zk.col(i) = zk;
75  }
76  // New state and its associated observation
77  thetak = mean(Thetak, 1);
78  mat zkMean = mean(Zk, 1); // Only for simplex case (other must use the weighted mean)
79  error = zkhat - zkMean;
80 
81  // Update covariance matrixes
82  LTheta = Thetak * Dsigma;
83  HL = Zk * Dsigma;
84  U = Pa + HL.t() * (Wi * HL);
85 
86  invU = inv(U);
87 
88  // Compute new estimate
89  Theta = thetak + LTheta * invU * HL.t() * (Wi * error);
90 
91  delete[] thetakdata;
92  delete[] xkdata;
93  delete[] zkdata;
94 
95  return norm(error, 2);
96 }
