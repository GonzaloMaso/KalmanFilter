32 AbstractROUKF* ConfigurationFileReader::getInstance() {

34  if(roukfModel)
35  return roukfModel;

37  Config config;
38 
39  try {
40  config.readFile(filename.c_str());
41  } catch (const FileIOException &fioex) {
42  std::cerr << "I/O error while reading configuration file for ROUKF." << std::endl;
43  } catch (const ParseException &pex) {
44  std::cerr << "Parse error at " << pex.getFile() << ":" << pex.getLine()
45  << " - " << pex.getError() << std::endl;
46  }
47 
48  int typeROUKF = 0;
49  try {
50  typeROUKF = config.lookup("FilterType");
51  } catch (const SettingNotFoundException &nfex) {
52  cerr << "Configuration file must explicit the FilterType (\"MappedROUKF\" | \"ROUKF\")." << endl;
53  }
54 
55  try {
56  nStates = config.lookup("States");
57  nParameters = config.lookup("Parameters");
58  nObservations = config.lookup("Observations");
59  } catch (const SettingNotFoundException &nfex) {
60  cerr << "States, Parameters or Observations number is missing." << endl;
61  }
62 
63  vector<double> initialGuess;
64  vector<double> parameterUncertainty;
65  try {
66  Setting& sVectorDoubles = config.lookup("InitialGuess");
67  for (int i = 0; i < sVectorDoubles.getLength(); ++i) {
68  initialGuess.push_back((double) sVectorDoubles[i]);
69  }
70  Setting& sVectorDoubles2 = config.lookup("ParameterUncertainty");
71  for (int i = 0; i < sVectorDoubles2.getLength(); ++i) {
72  parameterUncertainty.push_back((double) sVectorDoubles2[i]);
73  }
74  } catch (const SettingNotFoundException &nfex) {
75  cerr << "Error while reading InitialGuess or ParameterUncertainty fields." << endl;
76  }
77 
78  vector<AbstractParameterMapper *> parameterMapping;
79  vector<int> parametersPerMapping;
80  try {
81  Setting& mapperList = config.lookup("ParameterMapping");
82  for (int i = 0; i < mapperList.getLength(); ++i) {
83  int mapType = (int) mapperList[i]["type"];
84  switch (mapType) {
85  case IDENTITY:
86  parametersPerMapping.push_back(mapperList[i]["numParam"]);
87  parameterMapping.push_back(new IdentityParameterMapper());
88  break;
89  case EXPONENTIAL:
90  parametersPerMapping.push_back((int) mapperList[i]["numParam"]);
91  parameterMapping.push_back(new ExponentialParameterMapper());
92  break;
93  case SIGMOIDAL:
94  parametersPerMapping.push_back(mapperList[i]["numParam"]);
95  parameterMapping.push_back(new SigmoidParameterMapper((double) (mapperList[i]["min"]), (double) (mapperList[i]["max"])));
96  break;
97  default:
98  parametersPerMapping.push_back(mapperList[i]["numParam"]);
99  parameterMapping.push_back(new IdentityParameterMapper());
100  break;
101  }
102  }
103  } catch (const SettingNotFoundException &nfex) {
104  cerr << "Error while reading ParameterMapping fields." << endl;
105  }
106 
107  vector<double> observationsUncertainty;
108  try {
109  Setting& sVectorDoubles = config.lookup("ObservationsValues");
110  for (int i = 0; i < sVectorDoubles.getLength(); ++i) {
111  observations.push_back((double) sVectorDoubles[i]);
112  }
113  Setting& sVectorDoubles2 = config.lookup("ObservationsUncertainty");
114  for (int i = 0; i < sVectorDoubles2.getLength(); ++i) {
115  observationsUncertainty.push_back((double) sVectorDoubles2[i]);
116  }
117  } catch (const SettingNotFoundException &nfex) {
118  cerr << "Error while reading ObservationsValues or ObservationUncertainty fields." << endl;
119  }
120 
121  SigmaPointsGenerator::SIGMA_DISTRIBUTION sigmaDistribution = SigmaPointsGenerator::SIMPLEX;
122  try {
123  int sigmaType = config.lookup("SigmaDistribution");
124  sigmaDistribution = static_cast<SigmaPointsGenerator::SIGMA_DISTRIBUTION>(sigmaType);
125  } catch (const SettingNotFoundException &nfex) {
126  cerr << "Error while reading SigmaDistribution field." << endl;
127  }
128 
129  double tol = 1E-5;
130  try {
131  tol = config.lookup("ConvergenceTol");
132  } catch (const SettingNotFoundException &nfex) {
133  cerr << "Error while reading ConvergenceTol field." << endl;
134  }
135 
136  int maxIt = 1000;
137  try {
138  maxIt = config.lookup("MaxIterations");
139  } catch (const SettingNotFoundException &nfex) {
140  cerr << "Error while reading MaxIterations field." << endl;
141  }
142 
143  switch (typeROUKF) {
144  case MODEL_ROUKF:
145  roukfModel = new ROUKF(nObservations, nStates, nParameters,
146  &(observationsUncertainty[0]), &(parameterUncertainty[0]),
147  sigmaDistribution);
148  break;
149  case MODEL_MAPPED_ROUKF:
150  roukfModel = new MappedROUKF(nObservations, nStates, nParameters,
151  observationsUncertainty, parameterUncertainty, sigmaDistribution,
152  new CompositeParameterMapper(parametersPerMapping, parameterMapping));
153  break;
154  default:
155  roukfModel = new ROUKF(nObservations, nStates, nParameters,
156  &(observationsUncertainty[0]), &(parameterUncertainty[0]),
157  sigmaDistribution);
158  break;
159  }
160  roukfModel->setParameters(&(initialGuess[0]));
161  roukfModel->setTolerance(tol);
162  roukfModel->setMaxIterations(maxIt);
163 
164  return roukfModel;
165 
166 }
