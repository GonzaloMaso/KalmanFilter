132 double MappedROUKF::executeStep(vector<double> zkhatc, forwardOp A, observationOp H) {
133 
134  // Matrixes
135  mat Thetak(nParameters, sigma.n_cols), Xk(nStates, sigma.n_cols), Zk(nObservations, sigma.n_cols);
136  mat HL;
137  mat invU = inv(U);
138  mat C = chol(invU);
139 
140  // Column vectors
141  mat thetak, xk, zk;
142  mat zkhat(&(zkhatc[0]), nObservations, 1);
143 
144  double *zkdata;
145  zkdata = new double[nObservations];
146 
147  for (unsigned int i = 0; i < sigma.n_cols; i++) {
148  // Sampling
149  mat s = sigma.col(i);
150  xk = X + LX * C.t() * s;
151  thetak = Theta + LTheta * C.t() * s;
152 
153  // Transform theta_k -kalman parameters- to problem values -problem parameters-
154 
155  // Propagate sigma point
156  // Copies xk to a primitive type in order to achieve armadillo independence in extern programs
157  vector<double> thetakdata(thetak.memptr(), thetak.memptr() + nParameters);
158  vector<double> xkdata(xk.memptr(), xk.memptr() + nStates);
159 
160  thetakdata = mapper->unmap(thetakdata);
161  (*A)(&(xkdata[0]), nStates, &(thetakdata[0]), nParameters);
162  thetakdata = mapper->map(thetakdata);
163 
164  thetak = mat(&(thetakdata[0]), nParameters, 1);
165  xk = mat(&(xkdata[0]), nStates, 1);
166 
167  // Perform observation
168  (*H)(xk.memptr(), nStates, zkdata, nObservations);
169  zk = mat(zkdata, nObservations, 1);
170  // Transform theta_k to assimilation values
171  // TODO Transform to data assimilation values
172 
173  Xk.col(i) = xk;
174  Thetak.col(i) = thetak;
175  Zk.col(i) = zk;
176  }
177  // New state and its associated observation
178  xk = mean(Xk, 1);
179  thetak = mean(Thetak, 1);
180  mat zkMean = mean(Zk, 1); // Only constant alpha
181  error = zkhat - zkMean;
182 
183  // Update covariance matrixes
184  LX = Xk * Dsigma;
185  LTheta = Thetak * Dsigma;
186  HL = Zk * Dsigma;
187  U = Pa + HL.t() * (Wi * HL);
188 
189  invU = inv(U);
190 
191  // Compute new estimate
192  X = xk + LX * invU * HL.t() * (Wi * error);
193  Theta = thetak + LTheta * invU * HL.t() * (Wi * error);
194 
195  delete[] zkdata;
196  prevError = currError;
197  currError = norm(error, 2);
198  ++currIt;
199 
200  return currError;
201 }
